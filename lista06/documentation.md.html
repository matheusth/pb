41 
42         int resultado[][] = new int[matrizA.length][matrizB[0].length];
43 
44         for (int linhaMA = 0; linhaMA < matrizA.length; linhaMA++) {
45

47                 for (int linhaMB = 0; linhaMB < matrizA[linhaMA].length; linhaMB++) {
