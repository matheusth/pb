19     private static boolean matrizContem(int[][] matriz, int n) {
20         for (int i = 0; i < matriz.length; i++) {
21            for (int j = 0; j < matriz[i].length; j++) {
22                if(matriz[i][j] == n) {
23                    return true;
24                }
25            }
26         }
27         return false;
28     }

30     public static int[][] lerMatriz(int lines, int cols, Scanner sc) {
31         int matriz[][] = new int[lines][cols];
32 
33         for (int i = 0; i < matriz.length; i++) {
34             for (int j = 0; j < matriz[i].length; j++) {
35                 System.out.printf("Digite o valor do elemento (%d, %d): ", i, j);
36                 matriz[i][j] = sc.nextInt();
37 
38             }
39         }
40 
41         return matriz;
